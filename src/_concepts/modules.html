AngularJS Guide: Modules 

    // Module Definition
    var theApp = angular.module('TheApp', ['Malt.SubModule','Malt.Another']);
    theApp
      .config(function(){
        console.log('Configuring Module ' + theApp.name)
      })
      .run(function(){
        console.log('Running Module ' + theApp.name)
      });
    
    // There may be more than one config or run - Block
    theApp.run(function(){
      console.log( 'Dependencies ' + theApp.requires)
    });
    
    // Module Definition
    angular.module('Malt.SubModule',['Malt.Another']);
    
    // Module Getter
    var subModule = angular.module('Malt.SubModule');
    subModule
      .config(function(){
        console.log('Configuring Module Malt.SubModule')
      })
      .run(function(){
        console.log('Running Module Malt.SubModule')
      });
    
    // Module Definition
    angular.module('Malt.Another',[])
      .config(function(){
        console.log('Configuring Module Malt.Another')
      })
      .run(function(){
        console.log('Running Module Malt.Another')
      });
